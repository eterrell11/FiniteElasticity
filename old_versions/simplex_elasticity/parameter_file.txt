# Parameter file for three-field problem

subsection Material properties
  set Young's modulus = 1000
  set Poisson's ratio = 0.4999
  set Density = 10.0
end
subsection Time
  set Timestep =  0.0000625
  set End time =  1.0
  set Save time = 0.025
  set Start time = 0.0
end
# Time integrator options: 
# 1 for forward Euler
# 2 for SSPRK2
# 3 for SSPRK3
# 4 for modified trapezoidal rule
subsection Numerical parameters
  set alpha = 0
  set beta  = 0
  set Time integrator order = 4
  set n_ref = 3
  set Momentum order = 1
  set Pressure order = 1
  set Tau_pJ = 0.02
end
subsection Simulation parameters
  set Body force = 0
  set Initial velocity = 0
  set Traction Magnitude = 1
end
